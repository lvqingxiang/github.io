验签流程

点击小程序图标进入小程序时，会携带一串拼接内容data，经过Base64.decode解码后数据格式为：{"merchNo":"","terminalNo":"","dateTime":"","bankNo":"","sensitiveData":"","mac":""}。调用后台validateMsg接口对解码数据进行信息验证，返回JSON数据，保存decodeData，调用appAPI.login进行POS小程序验签操作。验签成功则跳转业务界面。

@PostMapping(value = "/validateMsg")
public JSONObject validateMsg(@RequestBody JSONObject request) {
return miniServiceClient.validateMsg(request);
}

@FeignClient(value = "mini-service-client", url = "${mini.server.url}")
public interface MiniServiceClient {
@PostMapping(path = "/api/miniprogram/validateMsg")
  JSONObject validateMsg(RequestBody JSONObject request);
}
import dsbridge from 'dsbridge'
Vue.prototype.$dsBridge = dsbridge
Vue.prototype.$gval = {}
validateMsg(this.$gval).then(res => {
  this.$gval.encryptionData = res.decodeData
  this.getValidate()
})
getValidate() {
  let req = {
    type: 'encryptionRequest',
    program_id: xxx, // 小程序ID
    name: xxx, // 小程序名称
    pwd: Base64.encode(xxx), //无登录操作则参数可随意
    encryption_data: this.$gVal.encryptionData 
}
  let res = this.$dsBridge.call(appAPI.login, JSON.stringify(req))
  let result = JSON.parse(res)
  if (result.resp_code === 'S02') return // 授权未通过
  if (result.resp_code === '00') {
    this.$router.push(xxx) 
  }
}
