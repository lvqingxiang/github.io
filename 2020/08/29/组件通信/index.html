  <div>

</template>

export default {
  data() {
    return {
      text: 'child',
    }
  },
  methods: {
    onClick(param) {
      console.log(param) // 'param'
    },
  },
}
</script>

<template>
  <div>
    <p @click="onClick">{{ text }}</p>
  </div>
</template>
<script>
export default {
  props: {
    text: String,
  },
  methods: {
    onClick() {
      this.$emit('click', 'param') //提交变量
    },
  },
}
</script>

中央事件总线EventBus

适用于小型项目，代码可维护性低。

const EventBus = new Vue()

EventBus.$on(event, callback)监听事件，EventBus.$emit(event, params)注册(触发)事件，EventBus.$off(event, callback)停止监听事件。

注册的事件在组件销毁时必须进行卸载，否则会出现多次挂载多个响应。

中央事件总线.jpg
自定义方法
class EventBus {
  constructor() {
    this.listeners = {}
  }
  isString(event) {
    return event && Object.prototype.toString.call(event) === '[object String]'
  }
  isFunction(callback) {
    return (
      callback &&
      Object.prototype.toString.call(callback) === '[object Function]'
    )
  }
  isVaild(event, callback) {
    return this.isString(event) && this.isFunction(callback)
  }
  on(event, callback) {
    if (this.isVaild(event, callback)) {
      if (this.listeners[event] && !this.listeners[event].includes(callback)) {
        this.listeners[event].push(callback) // 注册过该事件但未注册回调函数，加入回调函数数组
      } else {
        this.listeners[event] = [callback] // 未注册过该事件，初始化数组
      }
    } else {
      throw new Error('参数不合法')
    }
  }
  emit(event, ...args) {
    if (this.isString(event) && this.listeners[event]) {
      this.listeners[event].forEach((callback) => {
        callback(...args) // 触发on监听事件的callback
      })
    } else {
      throw new Error('参数不合法')
    }
  }
  off(event, callback) {
    if (this.isVaild(event, callback)) {
      if (this.listeners[event]) {
        let index = this.listeners[event].indexOf(callback)
        if (index !== -1) {
          this.listeners[event].splice(index, 1)
        }
      } else {
        this.listeners[event].length = 0
      } // 只提供event，移除该event下所有callback。提供event + callback，移除该callback
    } else {
      throw new Error('参数不合法')
    }
  }
}
