商城项目

本文最后更新于：2020-08-30 19:34:19

前提准备

node版本：v12.18.3

vue-cli版本：4.5.4

系统环境：macOS 10.15.5

代码编辑器：Visual Studio Code 1.48.0

创建项目

// 第一种情况 1.vue create [项目名称]，默认三种选择，选择前两项 2.进入项目目录 3.运行项目
// 第二种情况 1.vue create [项目名称]，选择第三种，自定义preset 2.进入项目目录 3.运行项目
vue create mall
cd mall
npm run serve
项目配置参考图.jpg

在项目根目录下创建.eslintrc.js文件，对项目编写规范进行约束，防止Eslint运行报错。

// `.eslintrc.js`
module.exports = {
  root: true,
  env: {
    node: true,
  },
  extends: ['plugin:vue/vue3-essential', 'eslint:recommended', '@vue/prettier'],
  parserOptions: {
    parser: 'babel-eslint',
  },
  rules: {
    'no-console': process.env.NODE_ENV === 'production' ? 'warn' : 'off',
    'no-debugger': process.env.NODE_ENV === 'production' ? 'warn' : 'off',
    quotes: ['error', 'single'],
    semi: ['error', 'never'],
  },
}

设置vue.config.js文件，对项目进行设置

// `vue.config.js`
const path = require('path')
const resolve = dir => path.join(__dirname, dir)
module.exports = {
  devServer: {
    open: true // 浏览器自启动
  },
  chainWebpack: config => {
    config.resolve.alias // 设置路径别名
      .set('assets', resolve('@/assets'))
      .set('views', resolve('@/views'))
      .set('components', resolve('@/components'))
    config.plugin('html').tap(args => {
      args[0].title = 'mall' // 修改应用名称
      return args
    })
  }
}

设置vue文件模版

  • 安装plop插件
npm run --save-dev plop
  • 根目录下创建plopfile.js
// plopfile.js
module.exports = function(plop) {
  plop.setGenerator('vue基础模板', {
    description: '创建vue文件',
    prompts: [
      {
        type: 'input', // 交互类型
        name: 'name', // 参数名称
        message: '请输入文件名称' // 交互提示
      },
      {
        type: 'input',
        name: 'path',
        message: '请输入文件创建目录'
      }
    ],
    actions: [
      {
        type: 'add', // 动作类型
        path: 'src/views/{{ path }}/{{ name }}.vue', // '{{  }}' 双大括号内设置动态参数
        templateFile: 'plop-templates/views/index.hbs' // 模板文件地址， 使用hbs文件
      }
    ]
  })
}
