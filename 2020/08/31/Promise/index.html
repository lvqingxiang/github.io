解决回调地狱

Promise既是对象，也是构造函数。

Promise包含三个状态，pending/fulfilled/rejected。pending表示异步操作未完成，fulfilled表示异步操作成功，rejected表示异步操作失败。状态变化只可能两种，(resolve)：未完成pending到成功fulfilled；(reject)：未完成pending到失败rejected，且状态变化只会发生一次。

构造函数

let promise = new Promise(function(resolve, reject) {
  let res = doSomeRequest() // 一般为ajax请求
  if (res.code === 200) {
    resolve('request success')
  } else {
    reject('request fail')
  }
})
promise
  .then(value => {
    console.log(value)
  })
  .catch(error => {
    console.log(error)
  })
  .finally(_ => {
    console.log('finally end')
  })

Promise.resolve()

将现有对象转为Promise对象

Promise.resolve("foo") => new Promise(resolve => resolve("foo"))

  • 参数是Promise实例，将返回该实例。

  • 参数是thenable对象，将该对象转为Promise对象后，将立即执行thenable对象中的then方法。

  • 参数是不具有then方法的对象或数据类型不是对象，返回新的Promise对象，且状态为fulfilled。

  • 不带有参数，将直接返回状态为fulfilled的Promise对象。

let thenable = {
  then: function(resolve, reject) {
    resolve(22)
  }
} // thenable对象示例

Promise.reject()

Promise.reject(reason)返回新的Promise对象，且状态为rejected。传入参数将直接作为后续方法的参数。

Promise.reject("hello").catch((error) => console.log(error)) // hello

Promise.all()

用于将多个Promises实例，包装成一个新的Promise实例。如果参数不是Promise实例，将调用Promise.resolve做处理。

如果Promise参数中有一个rejected，则回调失败reject，失败原因记录为第一个失败的Promise结果，否则所有Promise参数成功时，返回值组成数组，传给回调函数resolve。

// 第一种情况
let promise1 = Promise.resolve(43)
let promise2 = 2
let promise3 = new Promise(function(resolve, reject) {
  resolve('hello')
})
Promise.all([promise1, promise2, promise3]).then(values => {
  console.log(values)
})

// 第二种情况
const promise1 = async1()
const promise2 = async2() 
Promise.all([promise1, promise2]).then(values => {
  console.log(values)
})

Promise.race()

与Promise.all不同的是，Promise.race中的实例，一旦有实例成功fulfilled或失败rejected，率先改变的Promise实例的返回值将传递给回调函数。

let promise1 = Promise.resolve(43)
let promise2 = 2
let promise3 = new Promise(function(resolve, reject) {
  resolve('hello')
})
Promise.race([promise1, promise2, promise3]).then(value => {
  console.log(value)
})

Promise.allSettled()

用于将多个Promise实例，包装成一个新的Promise实例。如果参数不是Promise实例，将调用Promise.resolve做处理。

当所有Promise都返回结果，不论是fulfilled还是rejected，才会调用回调函数。

有时我们不关心异步操作的结果，只关心异步操作是否结束，Promise.allSettled将很有效果，Promise.all无法实现。

let promise1 = Promise.resolve(43)
let promise2 = 2
let promise3 = new Promise(function(resolve, reject) {
  resolve('hello')
})
Promise.allSettled([promise1, promise2, promise3]).then(results => {
  results.forEach(result => console.log(result.status))
})

自定义Promise + Ajax

function getJSON(url) {
  const promise = new Promise(function(resolve, reject) {
    const handler = function() {
      if (this.readyState !== 4) {
        return
      }
      if (this.statue === 200) {
        resolve(response)
      } else {
        reject(new Error('error'))
      }
    }
    const client = new XMLHttpRequest()
    client.open('GET', url)
    client.onreadystatechange = handler
    client.responseType = 'json'
    client.setRequestHeader('Accept', 'application/json')
    client.send()
  })
  return promise
}

async/await

Generator语法糖。

代码层面上表现为同步，执行层面上为Promise异步。

async定义的函数会返回promise对象。

await只能定义在async函数中，await等待表达式，表达式可以是Promise对象，也可以是其他表达式。

async函数内await之前的语句将同步执行，await语句将暂停async函数，阻塞函数内await之后的语句执行。如果await + Promise 将阻塞并暂停async函数，等待promise对象的resolve状态返回，如果await + 普通表达式，将执行表达式并暂停async函数。

暂停async函数，并不影响其他执行函数，其他函数将按序执行。

// 第一种情况
async function a1() {
  console.log('a1 start')
  await new Promise((resolve, reject) => {
    setTimeout(() => {
      resolve()
      console.log('resolve')
    }, 1000)
  })
  console.log('a1 end ')
}

function a2() {
  console.log('this is from a2')
}

a1()
a2()

/*打印结果：
a1 start
this is from a2 => await + Promise暂停函数a1后，并不影响函数a2的执行
resolve
a1 end 
*/

// 第二种情况
async function a1() {
  console.log('a1 start')
  await a3()
  console.log('a1 end ')
}

function a2() {
  console.log('this is from a2')
}

function a3() {
  console.log('a3')
}
a1()
a2()

/* 打印结果：
a1 start
a3 => await + 普通表达式，立即执行表达式后暂停a1
this is from a2
a1 end 
*/

// 第三种情况
async function async1() {
  console.log('async1 start')
  await async2()
  console.log('async1 end')
}

async function async2() {
  console.log('async2')
}

console.log('script start')

setTimeout(function() {
  console.log('setTimeout')
