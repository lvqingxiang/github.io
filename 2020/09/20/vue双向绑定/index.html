观察者模式&发布订阅模式

观察者模式&发布订阅模式.jpg

观察者模式中，观察者Observer能够感知Subject，发布订阅模式，发布者Publisher和订阅者Subscriber不知道对方的存在，通过中间商Event Channel进行通信。

观察者模式大多数是同步的，发布订阅模式大多数是异步的。

观察者模式Observer直接订阅主体Subject，当主体激活，所有Observer都将收到通知。

发布订阅模式将订阅者Subscriber希望订阅的事件(分类别)注册到调度中心，当发布者Publisher发布事件，由调度中心统一调度，区分类别，通知订阅者。

两者都实现了对象间一对多的依赖关系，当一个对象发生改变，所有依赖它的对象都将收到通知。

// 观察者模式
class Observer {
  constructor() {}
  update() {
    console.log('当Subject发生改变，触发Observer更新')
  }
}
class Subject {
  constructor() {
    this.Observers = []
  }
  attach(observer) {
    this.Observers.push(observer)
  }
  notify() {
    this.Observers.forEach((observer) => observer.update())
  }
}
let ob = new Observer()
let sub = new Subject()
sub.attach(ob)
sub.notify()

// 发布订阅模式
class EventChannel {
  constructor() {
    this.subscribers = {}
  }
  subscribe(event, callback) {
    if (!this.subscribers[event]) {
      this.subscribers[event] = [callback]
    } else {
      this.subscribers[event].push(callback)
    }
  }
  publish(event, ...arg) {
    if (this.subscribers[event]) {
      this.subscribers[event].forEach((callback) => callback(...arg))
    }
  }
}

Vue2

当JavaScript对象传入vue实例作为data选项，vue将遍历该对象的所有属性，并使用Object.defineProperty将属性转为getter/setter。

这些getter/setter对用户不可见，但在内部能够让vue追踪依赖，在属性被访问和修改时通知变更。

每个组件实例都对应一个watcher实例，它会在组件渲染的过程中将接触过的数据的属性记录为依赖。当依赖项的setter触发时，会通知watcher，从而使它关联的组件重新渲染。

采用数据劫持和发布订阅模式实现数据双向绑定。

实现Observer监听器

Observer主要作用是使用Object.defineProperty实现对属性的劫持和监听。给对象的每个属性添加getter和setter，在属性触发getter时，添加watcher，在触发setter时，通知watcher。

实现Dep消息订阅器

Dep消息订阅器存储所有watcher，并且可以添加watcher或通知watcher。

实现watcher订阅者

watcher的update方法能够触发compile中绑定的回调，进行视图更新。

compiler解析器

解析模版指令，初始化渲染页面视图，对每个指令对应的节点绑定更新函数，添加监听数据的watcher。

// 监听器observe
function observe(obj) {
  if (!obj || typeof obj != 'Object') return
  Object.keys(obj).forEach((key) => {
    defineReactive(obj, key, obj[key])
  })
}
function defineReactive(obj, key, val) {
  observe(val) // 递归遍历
  let dep = new Dep()
  Object.defineProperty(obj, key, {
    get() {
      dep.depend()
      console.log(`${key}属性被获取，新值为：${newVal}`)
      return val
    },
    set(newVal) {
      console.log(`${key}属性被修改，新值为：${newVal}`)
      val = newVal
      dep.notify()
    },
  })
}

// 消息订阅器Dep
class Dep {
  constructor() {
    this.watchers = []
  }
  // 增加订阅者
  addSub(watcher) {
    this.subscribers.push(watcher)
  }
  // 判断是否增加订阅者
  depend() {
    if (Dep.target) {
      this.addSub(Dep.target)
    }
  }
  notify() {
    this.subscribers.forEach((watcher) => {
      watcher.update()
    })
  }
}
// 订阅者watcher
class Watcher {
  constructor(vm, exp, callback) {
    this.vm = vm // vue实例对象
    this.exp = exp
    this.callback = callback // 更新函数
    this.value = this.get()
  }
  update() {
    let value = this.vm.data[this.exp] // 获取最新数据
    let oldVal = this.value // 之前this.get()获取的旧数据
    if (value !== oldVal) {
      this.value = value
      this.callback(this.vm, value, oldValue) // 调用更新函数进行更新
    }
  }
  get() {
    Dep.target = this // 将Dep.target赋值为当前的实例化对象watcher
    let value = this.vm.data[this.exp] // 强制执行observe监听器的get函数
    Dep.target = null // 释放
    return value
  }
}
function Compile(el, vm) {
  this.vm = vm
  this.el = el
  if (el) {
    this.$frag = this.noteToFragment(el) // 将dom节点转化为fragment虚拟节点
    this.compileElement(this.$frag) // 编译解析模版
  }
  el.appendChild(this.$frag) // 将编译后的虚拟DOM添加到根节点
  return this.$frag
}
Compile.prototype = {
  nodeToFragment(node) {
    let fragment = document.createDocumentFragment()
    let child = node.firstChild
    while (child) {
      // 循环将dom节点添加到fragment虚拟节点中
      fragment.appendChild(child)
      child = child.firstChild
    }
    return fragment
  },
  compileElement(elFlag) {
    // 解析模版
  },
  compileText(node, exp) {
    this.updateView(node, this.vm[exp])
    new Watcher(this.vm, exp, (value) => {
      // 数据变化时执行回调函数，并更新视图
      this.updateView(node, value)
    })
  },
  updateView(node, value) {
    node.textContent = typeof value !== 'undefined' ? value : ''
  },
}

Vue3

defineProperty挟持的是对象的属性，当对象属性新增时，需要重新遍历对象，对新增属性再使用defineProperty进行劫持。defineProperty无法监控到数组下标的变化，导致通过数组下标添加元素，不能实时响应，但push/pop/shift/unshift/splice/sort/reverse可以监听。

Proxy直接监听对象，而不是对象属性，不需要遍历操作，并返回一个新对象。Proxy还可以代理数组，还可以代理动态增加的属性。

Proxy返回新对象，可以直接操作新对象达到目的，defineProperty只能遍历对象属性直接修改。
