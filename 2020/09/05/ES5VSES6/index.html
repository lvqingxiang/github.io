includes/startsWith/endsWith

包含两个参数str，index

str：待检测字符串。

index：includes/startsWith表示字符串范围是[index - (length-1)]，endsWith表示字符串范围是[0 - index]。

let s = 'hello world'
// includes
console.log(s.includes('hello')) // true
console.log(s.includes('hello', 2)) // false
//startsWith
console.log(s.startsWith('h')) // true
console.log(s.startsWith('h', 3)) // false
//endsWith
console.log(s.endsWith('d')) // true
console.log(s.endsWith('d', 6)) // false

padStart/padEnd

用于补全字符串。

包含两个参数num，str

num：字符串补全生效的最大长度。

str：用来补全的字符串。

/*逻辑伪代码
原字符串oldStr 补全字符串str 最大长度num
*/
if(str.length >= num) {
  return oldStr // 补全字符串长度超出最大长度，返回原字符串
}else if(str.length + oldStr.length > num) {
  return str.substr(0, num - oldStr.length) + oldStr // padStart: 长度之和超出最大长度，截取补全字符串长度
  return oldStr + str.substr(0, num - oldStr.length) // padEnd: 长度之和超出最大长度，截取补全字符串长度
}else {
  str = str || '' // str默认使用空格
  return str.repeat(num - oldStr.length) + oldStr // padStart: 省略str则默认使用空格进行填充
  return oldStr + str.repeat(num - oldStr.length) // padEnd: 省略str则默认使用空格进行填充
}
// 补全数字
let num = '1800'
console.log(num.padEnd(10, '0')) // 180000000000
// 提示字符串格式
let date = '09-05'
console.log(date.padStart(10, 'YYYY-MM-DD')) // YYYY-09-05

trimStart/trimEnd

trimStart消除头部空格，保留尾部空格。trimEnd消除尾部空格，保留头部空格。

Number()

将任意类型值转换为数值。

Number接收对象或基本数据类型。

字符串：不能被转换的字符串将返回NaN

布尔值：true 为1，false为0

undefined：NaN

Null：0

对象：NaN，包含单个数值的数组将返回该数值

Math.trunc()

去除一个数的小数部分，返回整数部分。

非数值，使用Number()函数进行转换。

Math.sign()

判断一个数到底是正数、负数、还是零。

非数值，使用Number()函数进行转换。

// 伪代码
num > 0, return +1 // 正数返回+1
num < 0, return -1 // 负数返回-1
num === 0, return 0 // 0返回0
num === -0 return -0 // -0返回-0
other, return NaN // 其他返回NaN

Array.from()

用于将array-like对象和iterable对象转化为真正的数组。

支持map处理功能。

Array.from(arrayLike, item => item * 2)

Array.prototype.flat()

数组的成员还是数组时，可以使用Array.prototype.flat进行拉平。

默认拉平1层，可以传递整数参数，自定义拉平层数，或Infinity全部拉平。将跳过数组空位。

let arr = [[1], [[1], 2], [[[1], 2], 3], , ,]
console.log(arr.flat()) // [ 1, [ 1 ], 2, [ [ 1 ], 2 ], 3 ] 默认拉平1层，跳过空位
console.log(arr.flat(3)) // [ 1, 1, 2, 1, 2, 3 ] 自定义拉平层数，跳过空位
console.log(arr.flat(Infinity)) // [ 1, 1, 2, 1, 2, 3 ] // Infinity全部拉平，跳过空位
console.log(arr.flatMap((x) => x)) // [ 1, [ 1 ], 2, [ [ 1 ], 2 ], 3 ] map操作后flat，只能展开1层

数组常用方法

push/pop

push()在数组末尾添加1-n个新元素，返回数组长度，会修改原数组。

pop()删除数组末尾元素，返回该元素，会修改原数组。

shift/unshift

shift()删除数组头元素，返回该元素，会修改原数组。

unshift()在数组数组头部添加1-n个新元素，返回数组长度，会修改原数组。

join/concat

join将数组内容以指定分隔符串联为字符串，默认以逗号串联。

concat用于数组合并，返回新数组，原数组不变。

reverse/slice

reverse用于数组翻转，会修改原数组。

slice提取部分数组。返回新数组，原数组不变。

arr.slice(start, end)参数为start，end。start默认值为0，end默认值为arr.length - 1。

当slice参数为空，视为浅拷贝，提取范围为[start, end)。

可用于将类数组对象转化为真正的数组Array.prototype.slice.call(arrayLike)。

  begin = begin >= 0 ? begin : Math.max(0, this.length + begin)

splice/sort

splice用于删除或新增数组元素，返回被删除元素，会修改原数组。

arr.splice(start, count, ...args)，参数分别是起始位置，删除个数，插入元素。

sort默认按照字典序对数组进行排序

// splice
// 第一种方式
Array.prototype.mySplice = function (start = 0, count = this.length - start, ...args) {
  let result = []
  // 初始化
  start = start < 0 ? Math.max(start + this.length, 0) : Math.min(this.length, start)
  count = Math.min(Math.max(count, 0), this.length - start)
  if (start === 0) {
    return result
  } else {
    result = new Array(count)
    for (let i = 0; i < result.length; i++) {
      result[i] = this[start + i]
    }
    let argNum = args.length
    if (count > argNum) {
      for (let i = start + count; i < this.length; i++) {
        let fromIndex = i
        let toIndex = i - (count - argNum)
        this[toIndex] = this[fromIndex]
      }
      this.length = this.length - count + argNum
    }
    if (count < argNum) {
      for (let i = this.length - 1; i >= start + count; i--) {
        let fromIndex = i
        let toIndex = i + argNum - count
        this[toIndex] = this[fromIndex]
      }
    }
    for (let i = 0; i < argNum; i++) {
      this[start + i] = args[i]
    }
    return result
  }
}
var arr = [1, 2, 34, 6, 8]
console.log(arr.mySplice(1, 2, 4, 5, 7), arr) // [ 2, 34 ] [ 1, 4, 5, 7, 6, 8 ]

// 第二种方式
Array.prototype.mySplice = function (start = 0, count = this.length - start, ...args) {
  let result = []
  // 初始化
  start = start < 0 ? Math.max(start + this.length, 0) : Math.min(this.length, start)
  count = Math.min(Math.max(count, 0), this.length - start)
  if (start === 0) {
    return result
  } else {
    let leftArr = []
    let rightArr = []
    for (let i = 0; i < this.length; i++) {
      if (i < start) {
        leftArr.push(this[i])
      } else if (i >= start + count) {
        rightArr.push(this[i])
      } else {
        result.push(this[i])
      }
    }
    this.length = 0
    ;[...leftArr, ...args, ...rightArr].forEach((ele, i) => (this[i] = ele))
    return result
  }
}
var arr = [1, 2, 34, 6, 8]
console.log(arr.mySplice(1, 3, 4, 5, 7), arr) // [ 2, 34, 6 ] [ 1, 4, 5, 7, 8 ]

map/forEach/filter

map操作后返回新数组，且会跳过数组空位。

forEach遍历数组，且无法中断，且会跳过空位。

filter用于过滤数组成员，满足条件的数组成员组成新数组返回，原数组不变。

reduce/reduceRight

reduce遍历数组返回数组处理结果。reduce(callback(accumulator, curValue, curIndex, srcArray), initValue)，当initValue为空，将使用数组的第一个元素作为初始值。当initValue不为空，curIndex将从0开始累计。

reduce从左到右计算，reduceRight从右到左计算。

var sum = [0, 1, 2, 3].reduce(function (accumulator, curValue) {
  return accumulator + curValue;
}, 0); // 6
