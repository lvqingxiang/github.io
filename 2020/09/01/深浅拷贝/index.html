深浅拷贝

本文最后更新于：2020-09-06 08:49:25

对象赋值为「传址」操作，两个变量将指向同一个存储空间。

浅拷贝只会将对象的属性依次复制，但不会递归复制，即浅拷贝只会拷贝对象的第一层属性。当目标对象的属性为基本数据类型时，将进行「传值」操作，当属性为对象类型时，将进行「传址」操作。

深拷贝不仅复制对象的第一层属性，还将递归的拷贝目标对象的所有属性。

赋值

let user = {
  age: 19, // 基本类型
  grade: {
    pre: 'hi',
    full: false
  }
}
let temp = user // 赋值
temp.age = 20 // 修改数据类型
temp.grade.pre = 'hello' // 修改对象类型
temp.sex = '男' // 新增基本类型
temp.grade.date = [1997, 3, 20] // 新增对象属性

console.log(temp)
console.log(user)
/*
{
  age: 20,
  grade: { pre: 'hello', full: false, date: [ 1997, 3, 20 ] },
  sex: '男'
}
{
  age: 20,
  grade: { pre: 'hello', full: false, date: [ 1997, 3, 20 ] },
  sex: '男'
}
*/

自定义方法

function shallowClone(source) {
  let target = {}
  for (let i in source) {
    target[i] = source[i]
  }
  return target
}
// ---方法验证---
let source = {
  num: 19,
  date: [1997, 3, 20],
  user: {
    phone: '88888',
    age: 20
  }
}
console.log(shallowClone(source))
/*
{ num: 19, date: [ 1997, 3, 20 ], user: { phone: '88888', age: 20 } }
*/

自定义方法

const mapTag = '[object Map]'
const setTag = '[object Set]'
const arrayTag = '[object Array]'
const objectTag = '[object Object]'
const argsTag = '[object Arguments]'

const deepTag = [mapTag, setTag, arrayTag, objectTag, argsTag]

function isObject(target) {
  const type = typeof target
  return type !== null && (type === 'Object' || type === 'function')
}
function getType(target) {
  return Object.prototype.toString.call(target)
}
function getInit(target) {
  const cor = target.constructor
  return new cor()
}
function clone(source, map = new Map()) {
  // 基本数据类型处理
  if (!isObject(source)) {
    return source
  }
  // 初始化
  let target
  const type = getType(source)
  if (deepTag.includes(getType(type))) {
    target = getInit(source)
  }

    return source
