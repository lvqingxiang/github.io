setInterval

写一个mySetInterVal(fn, a, b)，每次间隔a, a+b, a+2b的时间，然后写一个myClear，停止上面的mySetInterval

var timer = null
var count = 0
function mySetInterval(fn, a, b) {
  function subSetInterval() {
    timer = setTimeout(() => {
      fn()
      count++
      subSetInterval()
    }, a + count * b)
  }
  subSetInterval()
}
function myClear() {
  clearTimeout(timer)
}

数组排序

合并二维有序数组组成一维有序数组，归并排序的思路

function merge(arr1, arr2) {
  let i = 0,
    j = 0,
    len1 = arr1.length,
    len2 = arr2.length,
    res = []
  while (i < len1 && j < len2) {
    if (arr1[i] < arr2[j]) {
      res.push(arr1[i++])
    } else {
      res.push(arr2[j++])
    }
  }
  if (i < len1) res = res.concat(arr1.slice(i))
  if (j < len2) res = res.concat(arr2.slice(j))
  return res
}
function mergeSort(arr) {
  const len = arr.length
  if (len === 1) return arr
  let mid = Math.floor(len / 2)
  return merge(mergeSort(arr.slice(0, mid)), mergeSort(arr.slice(mid)))
}
function myMergeSort(arr) {
  let narr = arr.flat(1)
  return mergeSort(narr)
}

斐波那契数列

多种方式实现斐波那契数列

function fib(n) {
  if (n === 1 || n === 2) return n
  return fib(n - 1) + fib(n - 2)
}

function fib(n) {
  let pre = 1,
    cur = 1,
    result = 0
  for (let i = 3; i < n; i++) {
    result = pre + cur
    pre = cur
    cur = result
  }
  return result
}

function lengthOfLongestString(str) {
  let temp = ''
  let res = []
  for (let i = 0, j = 0; i < str.length; i++, j++) {
    if (!temp) temp += str[i]
    let index = temp.indexOf(str[i])
    if (index > -1) {
      res.push(temp)
      temp = temp.slice(index + 1) + str[i]
    } else {
      temp += str[i]
    }

    return str.length

Chrome

介绍Chrome的几个版本

Chrome浏览器提供4种发布版本，稳定版stable,测试版Beta,开发者版Dev,金丝雀版Canary。

金丝雀版只限用于测试。是功能代码最先进的版本，适合进阶用户安装备用，尝鲜最新功能，并不适合日常使用。

开发版每周更新一次，勉强满足日常使用，适合Web开发者用来测试新功能和网页，提前测试应用兼容性。

测试版每月更新一次，是正式发布前的最后测试版本，普通用户日常可使用。测试版用户提前4-6周体验新功能，可以发现特定版本可能存在的问题。

稳定版以测试版为基础，是稳定性最高的Chrome版本。

HTTP缓存

浏览器每次发起请求，先在本地缓存中查找结果以及缓存标识，根据缓存标识判断是否使用本地缓存。如果缓存有效，则使用本地缓存；否则向服务器发起请求并携带缓存标识。

根据是否需要向服务器发起http请求，将缓存过程划分为2个部分：

强制缓存和协商缓存。强制缓存优先于协商缓存。

强缓存：服务器通知浏览器一个缓存时间，在缓存时间内，下次请求则直接使用缓存，不在时间内，则实行协商缓存策略。

协商缓存：让客户端和服务器之间能实现缓存文件是否更新的验证，提升缓存复用率，将缓存响应信息中的ETag和Last-Modified。

HTTP缓存从第二次请求开始。

对于强制缓存来说，response header中会有2个字段标明失效规则（Expires(http1.0)/Cache-Control(http1.1)。Expires的值为服务端返回的到期时间，由于服务端和客户端可能存在的时间偏差，导致缓存命中误差，在HTTP1.1版本中使用Cache-Control替代。Cache-Control常见取值有：

private：客户端可以缓存。

public：客户端和代理服务器均可缓存。

max-age=xxx：缓存的内容将在xxx秒后失效

no-cache：需要使用协商缓存验证缓存数据

no-store：所有内容都不会被缓存，强缓存和协商缓存都不会被触发

强制缓存的弊端很明显。每次根据过期时间判断缓存是否过期。但是在到达过期时间后，文件有可能并未改变，但依然需要重新请求资源。协商缓存则可以作为补充方案。

当服务器返回资源文件时，在response header中返回文件在服务器上的最近修改时间Last-Modified，当缓存文件过期，浏览器在request header中带上参数If-Last-Modified=(上一次返回的Last-Modified)，服务器比较If-Last-Modified与文件上次修改时间，一致则返回304，浏览器仍然使用缓存，否则重新返回文件资源，同时更新Last-Modified。

但由于Last-Modified修改时间只能精确到秒，如果文件在一秒内多次修改，服务器并不能捕捉到该改动，浏览器将无法获取最新文件资源。同时如果服务器上文件被多次修改但文件内容不变，服务器也需要多次重新返回文件资源。

为了解决文件修改时间不精确带来的问题，引入ETag。ETag基于资源的内容编码生成一串标识字符串，内容不同ETag不同。当服务器返回资源文件时，在response header中返回文件唯一标识ETag，当缓存文件过期，浏览器在request header中带上参数If-None-Match=(上一次返回的ETag)，服务器比较If-None-Match与文件的ETag，一致则返回304，浏览器仍然使用缓存，否则重新返回文件资源，同时更新ETag。
